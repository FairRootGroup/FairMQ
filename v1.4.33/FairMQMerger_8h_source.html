65  FairMQPollerPtr poller(NewPoller(chans));

67  if (fMultipart) {

70 
71  // Loop over the data input channels.
72  for (int i = 0; i < numInputs; ++i) {
73  // Check if the channel has data ready to be received.

75  FairMQParts payload;

85  }
86  }
87  }
